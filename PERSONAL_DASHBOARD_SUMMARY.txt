✨ الميزات الجديدة:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. 👤 DASHBOARD مخصص لكل مستخدم
   ├── عرض إحصائيات المستخدم فقط
   ├── لا يرى بيانات المستخدمين الآخرين
   ├── إحصائيات فورية:
   │   ├── ✅ عدد المبيعات المكتملة
   │   ├── ⏳ عدد المبيعات قيد المعالجة
   │   ├── 📋 عدد المبيعات المعلقة
   │   └── ❌ عدد المبيعات المرفوضة
   ├── 📊 رسم بياني تفاعلي (دائرة توزيع الحالات)
   ├── 📋 جدول تفاصيل الحالات
   └── 📄 جدول آخر 10 طلبات

2. 🔐 صلاحيات إدارية شاملة
   ├── صفحة إدارة جديدة: "إدارة رؤية Dashboards"
   ├── تفعيل/تعطيل Dashboard لكل مستخدم بشكل فردي
   ├── عرض جميع المستخدمين وحالتهم
   ├── تحديد سريع (Select All / Deselect All)
   └── حفظ الإعدادات بنقرة زر واحدة

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📍 الوصول:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  CRM Dashboard → لوحتي الشخصية ⭐ (جديد)

  CRM Dashboard → إدارة رؤية Dashboards ⭐ (جديد)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🔒 الصلاحيات:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Administrator:
  ✅ عرض لوحته الشخصية
  ✅ إدارة رؤية Dashboards للجميع
  ✅ جميع الصلاحيات الأخرى

Shop Manager:
  ✅ عرض لوحته الشخصية
  ✅ إدارة رؤية Dashboards للجميع
  ✅ جميع الصلاحيات الأخرى

Contributor (مندوب مبيعات):
  ✅ عرض لوحته الشخصية فقط
  ✅ عرض طلباته الخاصة
  ✅ إنشاء مبيعات يدوية

📊 محتويات Dashboard الشخصي:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. بطاقات إحصائية (Status Cards)
   ┌──────────────────┐
   │ ✅ المكتملة: 42  │
   ├──────────────────┤
   │ ⏳ قيد العمل: 15 │
   ├──────────────────┤
   │ 📋 المعلقة: 8    │
   ├──────────────────┤
   │ ❌ المرفوضة: 3   │
   └──────────────────┘

2. رسم بياني دائري (Doughnut Chart)
   يعرض توزيع جميع حالات طلبات المستخدم

3. جدول التفاصيل
   يعرض كل حالة وعدد الطلبات بها

4. جدول آخر 10 طلبات
   - رقم الطلب
   - اسم العميل
   - حالة الطلب
   - المبلغ
   - التاريخ
   - زر عرض التفاصيل

كمندوب مبيعات (Contributor):
  1. اذهب إلى CRM Dashboard
  2. اختر "لوحتي الشخصية"
  3. اعرض إحصائيات طلباتك فقط
  4. ابحث في آخر 10 طلبات
  5. اضغط على أي طلب لعرض التفاصيل

كإدارة (Admin/Shop Manager):
  1. اذهب إلى CRM Dashboard
  2. اختر "إدارة رؤية Dashboards"
  3. شاهد جميع المندوبين
  4. اختر من يرى لوحته الشخصية
  5. اضغط "حفظ الإعدادات"
